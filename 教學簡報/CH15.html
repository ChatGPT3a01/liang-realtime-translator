Tavily 搜尋

一個 POST 就有搜尋結果
# app.py
def tavily_search(key, query):
    r = requests.post('https://api.tavily.com/search', json={
        'api_key':key, 'query':query, 'max_results':5,
        'include_answer':True}, timeout=20)
    r.raise_for_status()
    return r.json()   # 含 answer 與 results[title,url,content]

後端 /api/ask

搜尋 → 餵給 AI → 回答
@app.route('/api/ask', methods=['POST'])
def api_ask():
    d = request.get_json(force=True) or {}
    q = (d.get('question') or '').strip()
    tv = d.get('tavily_key') or os.getenv('TAVILY_API_KEY') or ''
    ctx, sources, searched = '', [], False
    if tv:
        try:
            res = tavily_search(tv, q)
            for it in res.get('results', []):
                ctx += f"- {it['title']}: {it['content']}\n"
                sources.append({'title':it['title'],'url':it['url']})
            searched = True
        except Exception: pass   # 失敗就退回純 AI
    key = os.getenv('GEMINI_API_KEY') or os.getenv('GOOGLE_API_KEY')
    system = f"You are a travel assistant. Answer in Traditional Chinese. Use the web context if provided."
    user = q if not ctx else f"Question: {q}\n\nWeb context:\n{ctx}"
    answer = providers.generate('gemini', key, '', '', system, user)   # 沿用 CH06 的供應商
    return jsonify({'ok':True,'answer':answer,'sources':sources,'searched':searched})

前端：問答框

問題 → 答案 + 來源
async function ask(){
  const q = askInput.value.trim(); if(!q) return;
  askAnswer.textContent = '思考中…';
  const res = await fetch('/api/ask', {method:'POST',
    headers:{'Content-Type':'application/json'},
    body: JSON.stringify({ question:q, tavily_key: cfg.tavilykey || '' })});
  const d = await res.json();
  askAnswer.textContent = d.answer;
  askSources.innerHTML = d.sources.map(s=>`<a href="${s.url}" target="_blank">${s.title}</a>`).join('');
}
