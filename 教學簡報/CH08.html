後端 /api/tts

用 Gemini 產生語音（含重試）
# app.py — 雲端 TTS
from flask import Response
TTS_MODEL = 'gemini-3.1-flash-tts-preview'

@app.route('/api/tts', methods=['POST'])
def api_tts():
    from google.genai import types
    data = request.get_json(force=True, silent=True) or {}
    text = (data.get('text') or '').strip()
    if not text: return ('', 204)
    key = os.getenv('GEMINI_API_KEY') or os.getenv('GOOGLE_API_KEY')
    client = genai.Client(api_key=key)
    cfg = types.GenerateContentConfig(response_modalities=['AUDIO'],
        speech_config=types.SpeechConfig(voice_config=types.VoiceConfig(
            prebuilt_voice_config=types.PrebuiltVoiceConfig(voice_name='Kore'))))
    for _ in range(3):   # 預覽模型偶爾吐文字 → 重試
        try:
            r = client.models.generate_content(model=TTS_MODEL, contents=text, config=cfg)
            for p in r.candidates[0].content.parts:
                if getattr(p,'inline_data',None) and p.inline_data.data:
                    return Response(p.inline_data.data, mimetype='application/octet-stream')
        except Exception as e: last=str(e)
    return jsonify({'error':'tts failed'}), 400

前端 speak()

拿到語音、播放；失敗用瀏覽器
// app.js
async function speak(text, bcp){
  try{
    const res = await fetch('/api/tts', {method:'POST',
      headers:{'Content-Type':'application/json'}, body:JSON.stringify({text})});
    if(res.ok){ const buf = await res.arrayBuffer();
      if(buf.byteLength > 44){ playPCM(buf); return; } }
  }catch(e){}
  browserSpeak(text, bcp);   // 後備
}
function browserSpeak(text, bcp){
  const u = new SpeechSynthesisUtterance(text); u.lang = bcp;
  speechSynthesis.cancel(); speechSynthesis.speak(u);
}

播放雲端語音（Web Audio）

Gemini 回的是 24kHz PCM
// 播放 16-bit PCM
let playCtx=null, sources=[];
function playPCM(buf){
  if(!playCtx) playCtx=new (window.AudioContext||window.webkitAudioContext)({sampleRate:24000});
  if(playCtx.state==='suspended') playCtx.resume();   // 手機要解鎖
  const i16=new Int16Array(buf), f32=new Float32Array(i16.length);
  for(let i=0;i<i16.length;i++) f32[i]=i16[i]/32768;
  const b=playCtx.createBuffer(1,f32.length,24000); b.getChannelData(0).set(f32);
  const s=playCtx.createBufferSource(); s.buffer=b; s.connect(playCtx.destination);
  s.start(); sources.push(s);
}
function stopSpeak(){ sources.forEach(s=>{try{s.stop()}catch(e){}}); sources=[]; speechSynthesis.cancel(); }

手機的坑：音訊要「手勢解鎖」

iOS 特別嚴

手機（尤其 iPhone）要求音訊必須由「使用者點擊」啟動。因為 speak() 先 await 抓語音，等抓完手勢就過期了，可能靜音。

✅ 解法：在點擊當下先 playCtx.resume() 解鎖，並在 playPCM 裡也 resume。這樣手機才會出聲。

接上：翻完自動朗讀

在翻譯成功後呼叫 speak
// 翻譯成功後加一行
const out = await translate(text, NAME[langA.value], NAME[langB.value]);
result.textContent = out;
speak(out, BCP[langB.value]);   // ← 用目標語言朗讀

要「停止」時呼叫 stopSpeak()。可加一顆停止鈕，或朗讀鈕做成切換。
