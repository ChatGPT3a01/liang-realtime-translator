提示詞

主來源 + 備援
做 /api/currency：收 base、target、amount。
先打 open.er-api.com/v6/latest/{base} 取匯率，
拿不到再打 frankfurter.app 當備援。
回傳 {rate, result, date}。前端做換算器 UI，
選幣別或改金額就即時換算。

後端 /api/currency

主 ER-API、備援 Frankfurter
# app.py
import requests
@app.route('/api/currency', methods=['POST'])
def api_currency():
    d = request.get_json(force=True) or {}
    base=(d.get('base') or 'USD').upper(); target=(d.get('target') or 'TWD').upper()
    amount=float(d.get('amount') or 1)
    if base==target: return jsonify({'ok':True,'rate':1,'result':amount})
    try:
        r=requests.get(f'https://open.er-api.com/v6/latest/{base}', timeout=8).json()
        rate=(r.get('rates') or {}).get(target)
        if rate: return jsonify({'ok':True,'base':base,'target':target,
            'rate':rate,'result':amount*rate,'date':r.get('time_last_update_utc','')})
    except Exception: pass
    r=requests.get('https://api.frankfurter.app/latest',
        params={'from':base,'to':target}, timeout=8).json()   # 備援
    rate=(r.get('rates') or {}).get(target)
    return jsonify({'ok':bool(rate),'rate':rate,'result':(amount*rate) if rate else None})

前端：換算器

改金額或幣別就即時算
async function convert(){
  const amount = +document.getElementById('amt').value || 0;
  const from = curFrom.value, to = curTo.value;
  const res = await fetch('/api/currency', {method:'POST',
    headers:{'Content-Type':'application/json'},
    body: JSON.stringify({base:from, target:to, amount})});
  const d = await res.json();
  if(!d.ok || d.result == null){ curResult.textContent = '查不到此幣別匯率'; return; }
  curResult.textContent = `${amount} ${from} = ${d.result.toFixed(2)} ${to}`;
}
amt.addEventListener('input', convert);   // 打字就換算（可加防抖）
